  def extract_tags(text: str) -> list[str]:
      """Extract hashtags from text (e.g., #urgent, #todo, #important, #code-review)."""
      tags = re.findall(r"#([\w-]+)", text)
      seen = set()
      unique_tags = []
      for tag in tags:
          if tag not in seen:
              seen.add(tag)
              unique_tags.append(tag)
      return unique_tags

  def extract_action_items_with_tags(text: str) -> list[dict[str, Any]]:
      """Extract action items and their associated tags from text."""
      lines = text.splitlines()
      action_items = []

      for line in lines:
          stripped_line = line.strip()
          is_bullet = stripped_line.startswith("- ") or stripped_line.startswith("* ")

          if is_bullet or stripped_line.endswith("!") or stripped_line.lower().startswith("todo:"):
              if is_bullet:
                  item_text = stripped_line[2:]
              else:
                  item_text = stripped_line

              tags = extract_tags(line)

              # 移除标签
              for tag in tags:
                  item_text = item_text.replace(f"#{tag}", "").strip()

              # 移除 todo: 前缀
              if item_text.lower().startswith("todo:"):
                  item_text = item_text[5:].strip()

              # 移除尾随 "!"
              if item_text.endswith("!"):
                  item_text = item_text[:-1].strip()

              # 清理多余空格
              item_text = " ".join(item_text.split())

              if item_text:
                  action_items.append({
                      "description": item_text,
                      "tags": tags,
                  })

      return action_items

  ## Inputs
  - `$ARGUMENTS`: Optional pytest arguments (marker, path, or specific test name)

  ## Steps

  2. Analyze results:
    - If tests FAIL: Summarize failures and suggest next steps

  4. Output summary:
    - Test results (passed/failed/skipped)
    - Coverage percentage
    - Files with missing coverage
    - Recommendations for improving coverage
